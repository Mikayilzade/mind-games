  function ensureDailyPlan() {
    const date = todayKey();
    const existing = state.dailyPlans[date];
    if (existing && isValidPlan(existing)) return;

    const rng = mulberry32(hashString(date));
    const due = findDueRepeat();
    const selected = [];
    if (due) selected.push(due.gameId);
    for (const gameId of shuffle([...ACTIVE_GAMES], rng)) {
      if (!selected.includes(gameId)) selected.push(gameId);
      if (selected.length === 3) break;
    }

    const tasks = selected.map((gameId, index) => {
      if (due && index === 0 && gameId === due.gameId) {
        return {
          id: `${date}-repeat-${due.challengeId}`,
          gameId,
          challengeId: due.challengeId,
          spec: due.challengeSpec,
          isRepeat: true,
          repeatSourceResultId: due.id,
          completed: false,
          source: 'daily'
        };
      }
      const seed = hashString(`${date}-${gameId}-${index}-v3`);
      return {
        id: `${date}-${gameId}-${index}-v3`,
        gameId,
        challengeId: `${date}-${gameId}-${index}-v3`,
        spec: makeChallenge(gameId, seed, getAdaptiveLevel(gameId)),
        isRepeat: false,
        completed: false,
        source: 'daily'
      };
    });

    state.dailyPlans[date] = { date, estimatedMinutes: 9, tasks };
    saveState();
  }

  function isValidPlan(plan) {
    return Array.isArray(plan?.tasks)
      && plan.tasks.length === 3
      && plan.tasks.every(task => ACTIVE_GAMES.includes(task.gameId) && !isLegacySudokuTask(task));
  }

  function getTodayPlan() {
    ensureDailyPlan();
    return state.dailyPlans[todayKey()];
  }

  function findDueRepeat() {
    const today = todayKey();
    const candidates = state.results.filter(result => {
      if (!ACTIVE_GAMES.includes(result.gameId)) return false;
      if (!isCompatibleSpec(result.gameId, result.challengeSpec)) return false;
      const age = dayDifference(result.date, today);
      if (age < 5 || age > 30) return false;
      const repeatedRecently = state.results.some(other =>
        other.challengeId === result.challengeId
        && other.id !== result.id
        && dayDifference(other.date, today) < 5
      );
      return !repeatedRecently;
    });
    if (!candidates.length) return null;
    candidates.sort((a, b) => Math.abs(dayDifference(a.date, today) - 8) - Math.abs(dayDifference(b.date, today) - 8));
    return candidates[0];
  }

  function isCompatibleSpec(gameId, spec) {
    if (!spec) return false;
    if (gameId === 'grid') return Number.isInteger(spec.cols) && Array.isArray(spec.targets);
    if (gameId === 'pairs') return Array.isArray(spec.symbols) && Number.isInteger(spec.cols);
    if (gameId === 'sequence') return Array.isArray(spec.sequence);
    if (gameId === 'sudoku') return spec.size === 9 && Array.isArray(spec.puzzle) && spec.puzzle.length === 81;
    return false;
  }

  function getAdaptiveLevel(gameId) {
    const results = state.results.filter(r => r.gameId === gameId);
    if (!results.length) return 1;
    const last = results.slice(-4);
    const avg = last.reduce((sum, r) => sum + Number(r.accuracy || 0), 0) / last.length;
    const base = Math.min(7, 1 + Math.floor(results.length / 3));
    return Math.max(1, Math.min(7, avg >= 85 ? base + 1 : avg < 60 ? base - 1 : base));
  }

  function makeChallenge(gameId, seed, level) {
    const rng = mulberry32(seed);
    if (gameId === 'grid') {
      const cols = level >= 5 ? 5 : 4;
      const total = cols * cols;
      const count = Math.min(total - 4, 3 + Math.floor(level * 0.8));
      return { seed, cols, targets: sampleIndices(total, count, rng), previewMs: Math.max(1050, 2250 - level * 130) };
    }
    if (gameId === 'pairs') {
      const pairCount = Math.min(8, 3 + Math.floor(level / 2));
      const chosen = shuffle([...SYMBOLS], rng).slice(0, pairCount);
      return {
        seed,
        cols: pairCount <= 3 ? 3 : pairCount <= 6 ? 4 : 4,
        symbols: shuffle([...chosen, ...chosen], rng),
        previewMs: Math.max(1700, 3300 - level * 160)
      };
    }
    if (gameId === 'sequence') {
      const length = Math.min(10, 3 + level);
      return {
        seed,
        sequence: Array.from({ length }, () => SEQUENCE_SYMBOLS[Math.floor(rng() * SEQUENCE_SYMBOLS.length)]),
        paceMs: Math.max(340, 720 - level * 42)
      };
    }
    if (gameId === 'sudoku') {
      const givensByLevel = [46, 44, 41, 38, 35, 32, 30];
      const targetGivens = givensByLevel[Math.max(0, Math.min(givensByLevel.length - 1, level - 1))];
      return generateSudokuChallenge(seed, targetGivens);
    }
    throw new Error(`Unknown game: ${gameId}`);
  }
