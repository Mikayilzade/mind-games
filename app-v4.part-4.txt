      statusNode.classList.toggle('is-error', type === 'error');
      statusNode.classList.toggle('is-hint', type === 'hint');
    }

    function finishSudoku() {
      const correct = values.reduce((sum, value, index) => sum + (value === spec.solution[index] ? 1 : 0), 0);
      const empties = spec.puzzle.filter(value => value === 0).length;
      const wrongFinal = 81 - correct;
      activeGame.accuracy = Math.max(0, Math.round((empties - activeGame.mistakes - wrongFinal) / Math.max(1, empties) * 100));
      const hintPenalty = hintsUsed * 80;
      const undoPenalty = Math.min(120, undos * 3);
      activeGame.score = Math.max(100, Math.round(1500 + (81 - spec.clues) * 12 - elapsedSeconds() * 3 - activeGame.mistakes * 90 - hintPenalty - undoPenalty));
      activeGame.stats = { pencilMarks, undos, hintsUsed, wrongEntries };
      setTimeout(finishActiveGame, 300);
    }

    const boardClick = event => {
      const cell = event.target.closest('[data-sudoku-cell]');
      if (!cell) return;
      selected = Number(cell.dataset.sudokuCell);
      hintTarget = -1;
      renderBoard();
    };
    const controlClick = event => {
      const number = event.target.closest('[data-sudoku-number]');
      if (number) { enterNumber(Number(number.dataset.sudokuNumber)); return; }
      if (event.target.closest('[data-pencil]')) { togglePencil(); return; }
      if (event.target.closest('[data-undo]')) { undo(); return; }
      if (event.target.closest('[data-erase]')) { eraseSelected(); return; }
      if (event.target.closest('[data-hint]')) { useHint(); return; }
    };
    const keydown = event => {
      if (!activeGame || activeGame.gameId !== 'sudoku') return;
      if ((event.ctrlKey || event.metaKey) && event.key.toLowerCase() === 'z') { event.preventDefault(); undo(); return; }
      if (event.key === 'ArrowLeft') { event.preventDefault(); moveSelection(-1, 0); return; }
      if (event.key === 'ArrowRight') { event.preventDefault(); moveSelection(1, 0); return; }
      if (event.key === 'ArrowUp') { event.preventDefault(); moveSelection(0, -1); return; }
      if (event.key === 'ArrowDown') { event.preventDefault(); moveSelection(0, 1); return; }
      if (/^[1-9]$/.test(event.key)) { event.preventDefault(); enterNumber(Number(event.key)); return; }
      if (event.key === 'Backspace' || event.key === 'Delete' || event.key === '0') { event.preventDefault(); eraseSelected(); return; }
      if (event.key.toLowerCase() === 'p' || event.key === ' ') { event.preventDefault(); togglePencil(); }
    };

    boardNode.addEventListener('click', boardClick);
    stage.addEventListener('click', controlClick);
    document.addEventListener('keydown', keydown);

      boardNode.removeEventListener('click', boardClick);

      document.removeEventListener('keydown', keydown);
    };

  function sudokuPeers(index) {
    const row = Math.floor(index / 9), col = index % 9;
    const set = new Set();
    for (let i = 0; i < 9; i++) { set.add(row * 9 + i); set.add(i * 9 + col); }
    const boxRow = Math.floor(row / 3) * 3, boxCol = Math.floor(col / 3) * 3;
    for (let r = boxRow; r < boxRow + 3; r++) for (let c = boxCol; c < boxCol + 3; c++) set.add(r * 9 + c);
    set.delete(index);
    return [...set];
  }

  function isSudokuPeer(a, b) {
    if (a === b) return false;
    const ar = Math.floor(a / 9), ac = a % 9, br = Math.floor(b / 9), bc = b % 9;
    return ar === br || ac === bc || (Math.floor(ar / 3) === Math.floor(br / 3) && Math.floor(ac / 3) === Math.floor(bc / 3));
  }

  function findSudokuConflicts(values) {
    const conflicts = new Set();
    for (let i = 0; i < 81; i++) {
      if (!values[i]) continue;
      for (const peer of sudokuPeers(i)) if (peer > i && values[peer] === values[i]) { conflicts.add(i); conflicts.add(peer); }
    }
    return conflicts;
  }

  function conflictReason(a, b) {
    if (b == null) return 'в связанной области';
    const ar = Math.floor(a / 9), ac = a % 9, br = Math.floor(b / 9), bc = b % 9;
    if (ar === br) return 'в этой строке';
    if (ac === bc) return 'в этой колонке';
    return 'в этом блоке 3×3';
  }

  function sudokuCandidates(values, index) {
    if (values[index]) return [];
    const blocked = new Set(sudokuPeers(index).map(peer => values[peer]).filter(Boolean));
    return Array.from({ length: 9 }, (_, i) => i + 1).filter(number => !blocked.has(number));
  }

  function findBestHintCell(values) {
    let best = -1;
    let count = 10;
    for (let i = 0; i < 81; i++) {
      if (values[i]) continue;
      const candidates = sudokuCandidates(values, i);
      if (candidates.length && candidates.length < count) { best = i; count = candidates.length; }
    }
    return best;
  }

  function finishActiveGame() {
    if (!activeGame) return;
    clearInterval(timerInterval);
    if (activeGame.cleanup) activeGame.cleanup();
    const durationMs = Math.max(1000, performance.now() - activeGame.startTime);
    const task = activeGame.task;
    const previous = [...state.results].reverse().find(r => r.challengeId === task.challengeId);
    const previousCount = state.results.filter(r => r.challengeId === task.challengeId).length;
    const reviewCount = previousCount;
    const xpEarned = Math.max(12, Math.round(20 + activeGame.accuracy * .34 + Math.min(35, activeGame.score / 42)));
    const result = {
      id: `result-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
      date: todayKey(),

      challengeId: task.challengeId,
      challengeSpec: task.spec,
      durationMs: Math.round(durationMs),
      mistakes: activeGame.mistakes,
      accuracy: Math.max(0, Math.min(100, Math.round(activeGame.accuracy))),
      score: Math.round(activeGame.score),
      xpEarned,
      isRepeat: Boolean(task.isRepeat || previous),
      reviewCount,
      reviewDueDate: addDaysKey(todayKey(), reviewDelayDays(task.challengeId, reviewCount)),
      comparison: previous ? {
        previousResultId: previous.id,
        previousDurationMs: previous.durationMs,
        previousScore: previous.score,
        previousAccuracy: previous.accuracy,
        previousMistakes: previous.mistakes,
        previousStats: previous.stats || {}
      } : null,
      stats: activeGame.stats || {},
      actionLog: activeGame.actions || []
    };
    state.results.push(result);
    state.xp += xpEarned;
    markDailyTaskComplete(task.id, result.id);
    updateTrainingStreak(result.date);
    const newlyUnlocked = updateAchievements();
    saveState();
    renderResult(result, newlyUnlocked);
  }

  function renderResult(result, newlyUnlocked) {
    const stage = getGameStage();
    const previous = result.comparison;

  function markDailyTaskComplete(taskId, resultId) {

  function updateTrainingStreak(date) {
    if (state.lastTrainingDate === date) return;
    if (state.lastTrainingDate && dayDifference(state.lastTrainingDate, date) === 1) state.streak += 1;
    else state.streak = 1;
    state.bestStreak = Math.max(state.bestStreak, state.streak);
    state.lastTrainingDate = date;
  }

  function updateAchievements() {
    const unlocked = new Set(state.unlockedAchievements);
    const newly = [];
    for (const achievement of ACHIEVEMENTS) {
      if (!unlocked.has(achievement.id) && achievement.test(state)) {
        unlocked.add(achievement.id);
        newly.push(achievement);
      }
    }
    state.unlockedAchievements = [...unlocked];
    return newly;
  }

  function hasImprovement(s, percent) {
    const groups = new Map();
    for (const result of s.results) {
      if (!groups.has(result.challengeId)) groups.set(result.challengeId, []);
      groups.get(result.challengeId).push(result);
    }
    return [...groups.values()].some(items => items.length > 1 && items[items.length - 1].score >= items[0].score * (1 + percent / 100));
  }

  function getPlayerLevel(xp) { return Math.max(1, Math.floor(Math.sqrt(xp / 120)) + 1); }

  function reviewDelayDays(challengeId, count) { return 5 + Math.abs(hashString(`${challengeId}-${count}`)) % 6; }

  function formatDuration(ms) {
    const total = Math.max(0, Math.round(ms / 1000));
    const minutes = Math.floor(total / 60);
    const seconds = total % 60;
    return `${minutes}:${String(seconds).padStart(2, '0')}`;
  }

  function todayKey() { return dateKey(new Date()); }
  function dateKey(date) {
    const year = date.getFullYear();
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const day = String(date.getDate()).padStart(2, '0');
    return `${year}-${month}-${day}`;
  }
  function addDaysKey(key, days) {
    const date = new Date(`${key}T12:00:00`);
    date.setDate(date.getDate() + days);
    return dateKey(date);
  }
  function dayDifference(from, to) { return Math.round((new Date(`${to}T12:00:00`) - new Date(`${from}T12:00:00`)) / 86400000); }
  function lastNDays(n) {
    const days = [];
    const today = new Date();
    for (let i = n - 1; i >= 0; i--) { const date = new Date(today); date.setDate(today.getDate() - i); days.push(dateKey(date)); }
    return days;
  }
  function shortDay(key) { return ['вс', 'пн', 'вт', 'ср', 'чт', 'пт', 'сб'][new Date(`${key}T12:00:00`).getDay()]; }
  function formatDateLong(date) { return new Intl.DateTimeFormat('ru-RU', { weekday: 'long', day: 'numeric', month: 'long' }).format(date); }
  function formatDateShort(key) { return new Intl.DateTimeFormat('ru-RU', { day: 'numeric', month: 'short' }).format(new Date(`${key}T12:00:00`)); }

  function hashString(input) {
    let hash = 2166136261;
    for (let i = 0; i < String(input).length; i++) { hash ^= String(input).charCodeAt(i); hash = Math.imul(hash, 16777619); }
    return hash >>> 0;
  }
  function mulberry32(seed) {
    return function random() {
      let t = seed += 0x6D2B79F5;
      t = Math.imul(t ^ t >>> 15, t | 1);
      t ^= t + Math.imul(t ^ t >>> 7, t | 61);
      return ((t ^ t >>> 14) >>> 0) / 4294967296;
    };
  }
  function shuffle(array, rng = Math.random) {
    for (let i = array.length - 1; i > 0; i--) { const j = Math.floor(rng() * (i + 1)); [array[i], array[j]] = [array[j], array[i]]; }
    return array;
  }
  function sampleIndices(total, count, rng) { return shuffle(Array.from({ length: total }, (_, i) => i), rng).slice(0, count); }
  function escapeHtml(value) { return String(value).replace(/[&<>"]/g, char => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;' }[char])); }
