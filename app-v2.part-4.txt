        selected = Number(cell.dataset.sudokuCell);
        updateSudokuSelection(stage, values, spec.puzzle, selected);
        return;
      }

      const numberKey = event.target.closest('[data-number]');
      if (numberKey && selected >= 0 && editable[selected]) {
        const number = Number(numberKey.dataset.number);
        values[selected] = number;
        enteredCells.add(selected);
        renderSudokuCellValue(stage, selected, number, spec.puzzle);
        const conflict = sudokuConflict(values, selected);
        if (conflict) {
          activeGame.mistakes += 1;
          updateHudValue('mistakes', activeGame.mistakes);
          markSudokuConflict(stage, selected, conflict.indices);
          setSudokuMessage(stage, conflict.message, true);
        } else {
          clearSudokuErrors(stage);
          setSudokuMessage(stage, 'Число не конфликтует с текущей строкой, колонкой и блоком.');
        }
        updateSudokuSelection(stage, values, spec.puzzle, selected);
        if (values.every(Boolean)) checkSudoku(false);
        return;
      }

      if (event.target.closest('[data-clear-cell]') && selected >= 0 && editable[selected]) {
        values[selected] = 0;
        enteredCells.delete(selected);
        renderSudokuCellValue(stage, selected, 0, spec.puzzle);
        clearSudokuErrors(stage);
        updateSudokuSelection(stage, values, spec.puzzle, selected);
        return;
      }

      if (event.target.closest('[data-check-sudoku]')) {
        checkSudoku(true);
        return;
      }

      if (event.target.closest('[data-sudoku-hint]') && hintReady) {
        useSudokuHint();
      }
    };

    function checkSudoku(showIncompleteMessage) {
      clearSudokuErrors(stage);
      const wrong = [];
      values.forEach((value, index) => {
        if (value !== 0 && value !== spec.solution[index]) wrong.push(index);
      });
      if (wrong.length) {
        activeGame.mistakes += wrong.length;
        updateHudValue('mistakes', activeGame.mistakes);
        wrong.forEach(index => stage.querySelector(`[data-sudoku-cell="${index}"]`)?.classList.add('is-error'));
        setSudokuMessage(stage, `Неверных заполненных клеток: ${wrong.length}. Исправь отмеченные красным.`, true);
        return;
      }
      const empty = values.filter(value => value === 0).length;
      if (empty > 0) {
        if (showIncompleteMessage) setSudokuMessage(stage, `Пока всё верно. Осталось заполнить ${empty} клеток.`);
        return;
      }

      finished = true;
      const editableCount = editable.filter(Boolean).length;
      activeGame.accuracy = Math.max(0, Math.round((editableCount - activeGame.mistakes - activeGame.hintsUsed) / Math.max(1, editableCount) * 100));
      activeGame.score = Math.max(100, Math.round(1800 - elapsedSeconds() * 3 - activeGame.mistakes * 35 - activeGame.hintsUsed * 120 + (46 - spec.givens) * 25));
      setSudokuMessage(stage, 'Судоку решено правильно. Все строки, колонки и блоки заполнены без повторов.');
      setTimeout(finishActiveGame, 650);
    }

    function useSudokuHint() {
      let target = selected >= 0 && editable[selected] && values[selected] === 0 ? selected : -1;
      if (target < 0) {
        target = values.findIndex((value, index) => value === 0 && editable[index]);
      }
      if (target < 0) return;

      const candidates = getCandidates(values, target);
      const value = spec.solution[target];
      values[target] = value;
      enteredCells.add(target);
      activeGame.hintsUsed += 1;
      renderSudokuCellValue(stage, target, value, spec.puzzle);
      stage.querySelector(`[data-sudoku-cell="${target}"]`)?.classList.add('is-hint');
      selected = target;
      const row = Math.floor(target / 9) + 1;
      const col = target % 9 + 1;
      const reason = candidates.length === 1
        ? `Здесь подходит только ${value}: остальные числа уже встречаются в строке, колонке или блоке.`
        : `Подсказка: в клетке ${row}:${col} стоит ${value}. Возможные по текущим ограничениям числа были: ${candidates.join(', ') || '—'}.`;
      setSudokuMessage(stage, reason);
      updateSudokuSelection(stage, values, spec.puzzle, selected);
      if (values.every(Boolean)) checkSudoku(false);
    }

    stage.addEventListener('click', click);
    activeGame.cleanup = () => {
      clearInterval(hintCountdown);
      stage.removeEventListener('click', click);
    };
  }

  function sudokuCellMarkup(index, value, puzzle, selected) {
    const fixed = puzzle[index] !== 0;
    const row = Math.floor(index / 9);
    const col = index % 9;
    const classes = [
      'sudoku-cell',
      fixed ? 'is-fixed' : 'is-editable',
      index === selected ? 'is-active' : '',
      col === 2 || col === 5 ? 'block-right' : '',
      row === 2 || row === 5 ? 'block-bottom' : ''
    ].filter(Boolean).join(' ');
    return `<button class="${classes}" type="button" data-sudoku-cell="${index}" role="gridcell" aria-label="Строка ${row + 1}, колонка ${col + 1}${value ? `, число ${value}` : ', пусто'}">${value || ''}</button>`;
  }

  function renderSudokuCellValue(stage, index, value, puzzle) {
    const node = stage.querySelector(`[data-sudoku-cell="${index}"]`);
    if (!node) return;
    node.textContent = value || '';
    node.classList.toggle('has-value', Boolean(value) && puzzle[index] === 0);
    node.setAttribute('aria-label', `Строка ${Math.floor(index / 9) + 1}, колонка ${index % 9 + 1}${value ? `, число ${value}` : ', пусто'}`);
  }

  function updateSudokuSelection(stage, values, puzzle, selected) {
    const selectedValue = selected >= 0 ? values[selected] : 0;
    const row = selected >= 0 ? Math.floor(selected / 9) : -1;
    const col = selected >= 0 ? selected % 9 : -1;
    const blockRow = row >= 0 ? Math.floor(row / 3) : -1;
    const blockCol = col >= 0 ? Math.floor(col / 3) : -1;

    stage.querySelectorAll('[data-sudoku-cell]').forEach(node => {
      const index = Number(node.dataset.sudokuCell);
      const cellRow = Math.floor(index / 9);
      const cellCol = index % 9;
      const related = row >= 0 && (
        cellRow === row
        || cellCol === col
        || (Math.floor(cellRow / 3) === blockRow && Math.floor(cellCol / 3) === blockCol)
      );
      node.classList.toggle('is-related', related && index !== selected);
      node.classList.toggle('is-active', index === selected);
      node.classList.toggle('is-same-number', Boolean(selectedValue) && values[index] === selectedValue && index !== selected);
    });

    const candidatesNode = stage.querySelector('[data-candidates]');
    if (!candidatesNode) return;
    if (selected < 0 || puzzle[selected] !== 0) {
      candidatesNode.textContent = selectedValue ? `Выбрано число ${selectedValue}` : 'Выбери пустую клетку';
      return;
    }
    const candidates = getCandidates(values, selected);
    candidatesNode.innerHTML = candidates.length
      ? `Возможные без прямого конфликта: <strong>${candidates.join(' · ')}</strong>`
      : '<strong>Нет допустимых чисел.</strong> Проверь ранее заполненные клетки.';
  }

  function setSudokuMessage(stage, message, isError = false) {
    const node = stage.querySelector('[data-sudoku-message]');
    if (!node) return;
    node.textContent = message;
    node.classList.toggle('is-error-text', isError);
  }

  function clearSudokuErrors(stage) {
    stage.querySelectorAll('.sudoku-cell').forEach(node => node.classList.remove('is-error', 'is-conflict'));
  }

  function markSudokuConflict(stage, selected, indices) {
    clearSudokuErrors(stage);
    stage.querySelector(`[data-sudoku-cell="${selected}"]`)?.classList.add('is-error');
    indices.forEach(index => stage.querySelector(`[data-sudoku-cell="${index}"]`)?.classList.add('is-conflict'));
  }

  function sudokuConflict(values, index) {
    const value = values[index];
    if (!value) return null;
    const row = Math.floor(index / 9);
    const col = index % 9;
    const rowConflicts = [];
    const colConflicts = [];
    const blockConflicts = [];

    for (let c = 0; c < 9; c += 1) {
      const i = row * 9 + c;
      if (i !== index && values[i] === value) rowConflicts.push(i);
    }
    for (let r = 0; r < 9; r += 1) {
      const i = r * 9 + col;
      if (i !== index && values[i] === value) colConflicts.push(i);
    }
    const startRow = Math.floor(row / 3) * 3;
    const startCol = Math.floor(col / 3) * 3;
    for (let r = startRow; r < startRow + 3; r += 1) {
      for (let c = startCol; c < startCol + 3; c += 1) {
        const i = r * 9 + c;
        if (i !== index && values[i] === value) blockConflicts.push(i);
      }
    }

    const indices = [...new Set([...rowConflicts, ...colConflicts, ...blockConflicts])];
    if (!indices.length) return null;
    const places = [];
    if (rowConflicts.length) places.push('строке');
    if (colConflicts.length) places.push('колонке');
    if (blockConflicts.length) places.push('блоке 3×3');
    return { indices, message: `Число ${value} уже есть в ${places.join(', ')}.` };
  }

  function getCandidates(values, index) {
    if (values[index] !== 0) return [];
    const candidates = [];
    for (let value = 1; value <= 9; value += 1) {
      values[index] = value;
      if (!sudokuConflict(values, index)) candidates.push(value);
    }
    values[index] = 0;
    return candidates;
  }

  function generateSudokuChallenge(seed, targetGivens) {
    const rng = mulberry32(seed);
    const solution = makeSudokuSolution(rng);
    const puzzle = [...solution];
    const order = shuffle(Array.from({ length: 81 }, (_, index) => index), rng);
    let givens = 81;

    for (const index of order) {
      if (givens <= targetGivens) break;
      const previous = puzzle[index];
      puzzle[index] = 0;
      if (countSudokuSolutions(puzzle, 2) !== 1) {
        puzzle[index] = previous;
      } else {
        givens -= 1;
      }
    }

    return {
      seed,
      size: 9,
      blockRows: 3,
      blockCols: 3,
      givens,
      puzzle,
      solution
    };
  }

  function makeSudokuSolution(rng) {
    const base = Array.from({ length: 81 }, (_, index) => {
      const row = Math.floor(index / 9);
      const col = index % 9;
      return (row * 3 + Math.floor(row / 3) + col) % 9 + 1;
    });
    const digitMap = shuffle([1,2,3,4,5,6,7,8,9], rng);
    const rowBands = shuffle([0,1,2], rng);
    const colStacks = shuffle([0,1,2], rng);
    const rows = rowBands.flatMap(band => shuffle([0,1,2], rng).map(offset => band * 3 + offset));
    const cols = colStacks.flatMap(stack => shuffle([0,1,2], rng).map(offset => stack * 3 + offset));

    return rows.flatMap(row => cols.map(col => digitMap[base[row * 9 + col] - 1]));
  }

  function countSudokuSolutions(board, limit = 2) {
    const values = [...board];
    let count = 0;

    function solve() {
      if (count >= limit) return;
