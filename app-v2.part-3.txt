  }

  function closeGame() {
    if (activeGame?.cleanup) activeGame.cleanup();
    clearInterval(timerInterval);
    modalRoot.innerHTML = '';
    activeGame = null;
  }

  function startActiveGame() {
    if (!activeGame) return;
    activeGame.startTime = performance.now();
    if (activeGame.gameId === 'grid') startGridGame();
    if (activeGame.gameId === 'pairs') startPairsGame();
    if (activeGame.gameId === 'sequence') startSequenceGame();
    if (activeGame.gameId === 'sudoku') startSudokuGame();
  }

  function startGridGame() {
    const { cols, targets, previewMs } = activeGame.spec;
    const stage = getGameStage();
    let selected = [];
    let canTap = false;

    stage.innerHTML = gameLayout(
      'Запомни клетки',
      targets.length,
      Array.from({ length: cols * cols }, (_, index) =>
        `<button class="memory-cell ${targets.includes(index) ? 'is-lit' : ''}" data-cell="${index}" type="button" aria-label="Клетка ${index + 1}"></button>`
      ).join(''),
      'memory-grid',
      cols
    );
    stage.classList.add('is-memory-game');
    startHudTimer();

    const previewTimer = setTimeout(() => {
      stage.querySelectorAll('.memory-cell').forEach(cell => cell.classList.remove('is-lit'));
      stage.querySelector('.instruction').textContent = `Выбери ${targets.length} клеток`;
      canTap = true;
    }, previewMs);

    const click = event => {
      const cell = event.target.closest('[data-cell]');
      if (!cell || !canTap) return;
      const index = Number(cell.dataset.cell);
      if (selected.includes(index)) {
        selected = selected.filter(value => value !== index);
        cell.classList.remove('is-selected');
      } else if (selected.length < targets.length) {
        selected.push(index);
        cell.classList.add('is-selected');
      }
      if (selected.length !== targets.length) return;

      canTap = false;
      const correct = selected.filter(value => targets.includes(value)).length;
      const mistakes = targets.length - correct;
      stage.querySelectorAll('.memory-cell').forEach((node, index2) => {
        if (targets.includes(index2)) node.classList.add('is-correct');
        else if (selected.includes(index2)) node.classList.add('is-wrong');
      });
      activeGame.mistakes = mistakes;
      activeGame.accuracy = Math.round(correct / targets.length * 100);
      activeGame.score = Math.max(100, Math.round(500 + targets.length * 110 - elapsedSeconds() * 8 - mistakes * 90));
      setTimeout(finishActiveGame, 800);
    };

    stage.addEventListener('click', click);
    activeGame.cleanup = () => {
      clearTimeout(previewTimer);
      stage.removeEventListener('click', click);
    };
  }

  function startPairsGame() {
    const { symbols, cols, previewMs } = activeGame.spec;
    const stage = getGameStage();
    let open = [];
    const matched = new Set();
    let canTap = false;
    let moves = 0;

    stage.innerHTML = gameLayout(
      'Запомни расположение',
      symbols.length / 2,
      symbols.map((symbol, index) => `<button class="pair-card is-open" data-card="${index}" type="button"><span>${symbol}</span></button>`).join(''),
      'pairs-grid',
      cols
    );
    stage.classList.add('is-pairs-game');
    startHudTimer();

    const previewTimer = setTimeout(() => {
      stage.querySelectorAll('.pair-card').forEach(card => card.classList.remove('is-open'));
      stage.querySelector('.instruction').textContent = 'Открой все пары';
      canTap = true;
    }, previewMs);

    const click = event => {
      const card = event.target.closest('[data-card]');
      if (!card || !canTap) return;
      const index = Number(card.dataset.card);
      if (matched.has(index) || open.includes(index)) return;
      card.classList.add('is-open');
      open.push(index);

      if (open.length !== 2) return;
      moves += 1;
      updateHudValue('moves', moves);
      canTap = false;
      const [a, b] = open;

      if (symbols[a] === symbols[b]) {
        matched.add(a);
        matched.add(b);
        stage.querySelector(`[data-card="${a}"]`)?.classList.add('is-matched');
        stage.querySelector(`[data-card="${b}"]`)?.classList.add('is-matched');
        open = [];
        canTap = true;
        if (matched.size === symbols.length) {
          const optimal = symbols.length / 2;
          activeGame.mistakes = Math.max(0, moves - optimal);
          activeGame.accuracy = Math.max(0, Math.round(optimal / moves * 100));
          activeGame.score = Math.max(100, Math.round(700 + optimal * 110 - elapsedSeconds() * 6 - activeGame.mistakes * 45));
          setTimeout(finishActiveGame, 450);
        }
      } else {
        activeGame.mistakes += 1;
        setTimeout(() => {
          stage.querySelector(`[data-card="${a}"]`)?.classList.remove('is-open');
          stage.querySelector(`[data-card="${b}"]`)?.classList.remove('is-open');
          open = [];
          canTap = true;
        }, 560);
      }
    };

    stage.addEventListener('click', click);
    activeGame.cleanup = () => {
      clearTimeout(previewTimer);
      stage.removeEventListener('click', click);
    };
  }

  async function startSequenceGame() {
    const session = activeGame;
    const { sequence, paceMs } = session.spec;
    const stage = getGameStage();
    let input = [];
    let alive = true;

    stage.innerHTML = `
      <div class="game-hud"><div class="hud-chip"><span>Длина</span><strong>${sequence.length}</strong></div><div class="hud-chip"><span>Время</span><strong data-time>0:00</strong></div></div>
      <p class="instruction">Следи за последовательностью</p>
      <div class="sequence-display" data-sequence-display></div>
      <div class="sequence-pad" hidden>${SEQUENCE_SYMBOLS.map(symbol => `<button class="sequence-key" type="button" data-symbol="${symbol}">${symbol}</button>`).join('')}</div>`;
    stage.classList.add('is-sequence-game');
    startHudTimer();

    const display = stage.querySelector('[data-sequence-display]');
    for (let index = 0; index < sequence.length && alive && activeGame === session; index += 1) {
      display.innerHTML = `<span class="sequence-token">${sequence[index]}</span>`;
      await wait(paceMs);
      display.innerHTML = '';
      await wait(Math.max(140, paceMs * 0.35));
    }
    if (!alive || activeGame !== session) return;

    stage.querySelector('.instruction').textContent = 'Повтори цепочку';
    stage.querySelector('.sequence-pad').hidden = false;

    const click = event => {
      const key = event.target.closest('[data-symbol]');
      if (!key) return;
      const symbol = key.dataset.symbol;
      const expected = sequence[input.length];
      input.push(symbol);
      display.innerHTML = input.map(value => `<span class="sequence-token">${value}</span>`).join('');

      if (symbol !== expected) {
        activeGame.mistakes = 1;
        activeGame.accuracy = Math.round((input.length - 1) / sequence.length * 100);
        activeGame.score = Math.max(50, 250 + (input.length - 1) * 100);
        stage.querySelector('.instruction').textContent = `Нужен был символ ${expected}`;
        stage.querySelector('.sequence-pad').hidden = true;
        setTimeout(finishActiveGame, 720);
      } else if (input.length === sequence.length) {
        activeGame.accuracy = 100;
        activeGame.score = Math.max(100, Math.round(600 + sequence.length * 125 - elapsedSeconds() * 5));
        stage.querySelector('.sequence-pad').hidden = true;
        setTimeout(finishActiveGame, 420);
      }
    };

    stage.addEventListener('click', click);
    session.cleanup = () => {
      alive = false;
      stage.removeEventListener('click', click);
    };
  }

  function startSudokuGame() {
    const spec = activeGame.spec;
    const stage = getGameStage();
    const values = [...spec.puzzle];
    const editable = spec.puzzle.map(value => value === 0);
    let selected = editable.findIndex(Boolean);
    let hintReady = false;
    let finished = false;
    const enteredCells = new Set();

    stage.classList.add('is-sudoku-game');
    stage.innerHTML = `
      <div class="game-hud"><div class="hud-chip"><span>Ошибки</span><strong data-mistakes>0</strong></div><div class="hud-chip"><span>Время</span><strong data-time>0:00</strong></div></div>
      <div class="sudoku-info">
        <strong>${sudokuDifficultyLabel(spec.givens)} · открыто ${spec.givens} чисел</strong>
        <span>В каждой строке, колонке и блоке 3×3 должны быть числа 1–9 без повторов.</span>
      </div>
      <p class="instruction" data-sudoku-message>Выбери пустую клетку. Ниже появятся возможные числа.</p>
      <div class="sudoku-board" role="grid" aria-label="Судоку 9 на 9">
        ${values.map((value, index) => sudokuCellMarkup(index, value, spec.puzzle, selected)).join('')}
      </div>
      <div class="sudoku-candidates" data-candidates></div>
      <div class="sudoku-pad">
        ${[1,2,3,4,5,6,7,8,9].map(number => `<button class="sudoku-key" type="button" data-number="${number}">${number}</button>`).join('')}
        <button class="sudoku-key sudoku-action" type="button" data-clear-cell>⌫</button>
      </div>
      <div class="sudoku-actions">
        <button class="secondary-button" type="button" data-check-sudoku>Проверить</button>
        <button class="secondary-button" type="button" data-sudoku-hint disabled>Подсказка через 0:30</button>
      </div>`;

    startHudTimer();
    updateSudokuSelection(stage, values, spec.puzzle, selected);

    const hintCountdown = setInterval(() => {
      if (!activeGame || finished) return;
      const remaining = Math.max(0, 30 - Math.floor(elapsedSeconds()));
      const button = stage.querySelector('[data-sudoku-hint]');
      if (!button) return;
      if (remaining <= 0) {
        hintReady = true;
        button.disabled = false;
        button.textContent = 'Показать подсказку';
        clearInterval(hintCountdown);
      } else {
        button.textContent = `Подсказка через 0:${String(remaining).padStart(2, '0')}`;
      }
    }, 250);

    const click = event => {
      if (finished) return;
      const cell = event.target.closest('[data-sudoku-cell]');
      if (cell) {
